* Graph Validator
:PROPERTIES:
:mode: compile
:artifact_role: proposal
:algorithm: Proposal/Receipt

:maps_to: org/proposals.org, org/receipts.org, org/claims.org, org/closure.org
:END:

This block validates the promotion graph across all org files.

#+NAME: validate-promotion-graph
#+BEGIN_SRC emacs-lisp :tangle no
(defun org-promotion-graph-validate ()
  "Validate promotion edges across org files."
  (let ((files '("org/claims.org" "org/proposals.org" "org/closure.org" "org/receipts.org"))
        (errors nil))
    (dolist (file files)
      (with-temp-buffer
        (insert-file-contents file)
        (org-mode)
        (goto-char (point-min))
        (while (re-search-forward "^\\*+ " nil t)
          (when (org-at-heading-p)
            (let* ((props (org-entry-properties))
                   (step-id (cdr (assoc "step_identity" props))))
              (when step-id
                (cond
                 ((string-match "proposals\\." step-id)
                  (unless (assoc "derived_from" props)
                    (push (format "PROPOSAL missing derived_from: %s" step-id) errors))
                  (unless (assoc "promotes_to" props)
                    (push (format "PROPOSAL missing promotes_to: %s" step-id) errors)))
                 ((string-match "receipts\\." step-id)
                  (unless (assoc "receipt_for" props)
                    (push (format "RECEIPT missing receipt_for: %s" step-id) errors)))
                 ((string-match "closure\\." step-id)
                  (when (string= (cdr (assoc "status" props)) "closed")
                    (message "Closure %s would require receipt check" step-id))))))))))
    (if errors
        (progn (message "VALIDATION FAILED:\n%s" (mapconcat 'identity (reverse errors) "\n"))
               nil)
      (message "VALIDATION PASSED: Promotion graph consistent")
      t)))

(org-promotion-graph-validate)
#+END_SRC
